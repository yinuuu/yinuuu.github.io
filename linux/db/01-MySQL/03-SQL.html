9 子查询 #

shell
-- 子查询：查询中的嵌套查询 
-- 示例：查询年龄最高的人员信息 

-- 1 查询最高的年龄 
select max(age) from emp; 

-- 2 查询人员信息，并且年龄等于30的 
select * from emp where emp.`age` = 30; 

-- 一条sql就可以完成这个操作，这就是子查询 
select * from emp where emp.`salary` = (select max(age) from emp);

-- 子查询：查询中的嵌套查询 
-- 示例：查询年龄最高的人员信息 

-- 1 查询最高的年龄
